% -----------LAMBDA CALC SEMANTICS---------------
\section{Lambda Calculus Semantics}

\framed[]{
\begin{gather*}
  \llbracket \cdot \rrbracket \in \Conf \to \Value
\end{gather*}
}

\vsp

\framed {
  \infax[clo]
  {
    \tup[\fun] \eval \clot
  }
}

\vsp

\framed {
  \infrule[var]
  {
    \assign{v}{\env(x)}
  }
  {
    \tup[x] \eval v
  }
}

\vsp

\framed {
  \infrule[apply]
  {
    \tup[e_1] \eval \clot[\env'] \andalso \tup[e_2] \eval v
    \\
    \assign{\env''}{\env'[x \mapsto v]}
  }
  {
    \tup[e_1 \; e_2] \eval \tup[e][\env'']
  }
}
